void timer_init() {
  cli();                                             // Disable interrupts
  TCCR2A = (1 << WGM21);                             // Set CTC mode
  TCCR2B = (1 << CS22) | (1 << CS21) | (1 << CS20);  // 1024 Prescaler
  OCR2A = 155;                                       // (16MHz / 1024 / 100Hz) - 1 = 155.25
  TIMSK2 = (1 << OCIE2A);                            // Enable CTC interrupt
  sei();                                             // Enable interrupts
}

void motor_run(int speedDC_left, int speedDC_right) {
  // Left motor control
  if (speedDC_left >= 0) {
    digitalWrite(DIR_LEFT, LOW);  // Set direction forward
    analogWrite(PWM_LEFT, speedDC_left);
  } else {
    digitalWrite(DIR_LEFT, HIGH);  // Set direction backward
    analogWrite(PWM_LEFT, 255 + speedDC_left);
  }

  // Right motor control
  if (speedDC_right >= 0) {
    digitalWrite(DIR_RIGHT, LOW);  // Set direction forward
    analogWrite(PWM_RIGHT, speedDC_right);
  } else {
    digitalWrite(DIR_RIGHT, HIGH);  // Set direction backward
    analogWrite(PWM_RIGHT, 255 + speedDC_right);
  }
}
void motor_brake() {
  digitalWrite(DIR_LEFT, HIGH);  // Set direction forward
  digitalWrite(PWM_LEFT, HIGH);
  digitalWrite(DIR_RIGHT, HIGH);  // Set direction backward
  digitalWrite(PWM_RIGHT, HIGH);
  digitalWrite(Fan, LOW);  // turn off fan
}
void RGB(int color) {
  switch (color) {
    case 0:  // none
      {
        digitalWrite(RED, 1);
        digitalWrite(GREEN, 1);
        digitalWrite(BLUE, 1);
        break;
      }
    case 1:  // red
      {
        digitalWrite(RED, 0);
        digitalWrite(GREEN, 1);
        digitalWrite(BLUE, 1);
        break;
      }
    case 2:  // green
      {
        digitalWrite(RED, 1);
        digitalWrite(GREEN, 0);
        digitalWrite(BLUE, 1);
        break;
      }
    case 3:  // blue
      {
        digitalWrite(RED, 1);
        digitalWrite(GREEN, 1);
        digitalWrite(BLUE, 0);
        break;
      }

    case 4:  // purple
      {
        digitalWrite(RED, 0);
        digitalWrite(GREEN, 1);
        digitalWrite(BLUE, 0);
        break;
      }
    case 5:  // yellow
      {
        digitalWrite(RED, 0);
        digitalWrite(GREEN, 0);
        digitalWrite(BLUE, 1);
        break;
      }
    case 6:  // white
      {
        digitalWrite(RED, 0);
        digitalWrite(GREEN, 0);
        digitalWrite(BLUE, 0);
        break;
      }
  }
}
void PD_calculate() {
  unsigned char temp = 0;
  unsigned int sum = 0;
  unsigned long avg = 0;
  int error, iP, iD;
  int iRet;
  Sum_sensor_value = 0;
  sensorValue[0] = analogRead(A0);
  sensorValue[1] = analogRead(A1);
  sensorValue[2] = analogRead(A2);
  sensorValue[3] = analogRead(A3);
  sensorValue[4] = analogRead(A4);
  sensorValue[5] = analogRead(A5);
  sensorValue[6] = analogRead(A6);
  sensorValue[7] = analogRead(A7);
  for (int j = 0; j < 8; j++) {
    if (calibrating == 0) {
      if (sensorValue[j] < black_value[j])
        sensorValue[j] = black_value[j];
      if (sensorValue[j] > white_value[j])
        sensorValue[j] = white_value[j];
      sensorPDValue[j] = map(sensorValue[j], black_value[j], white_value[j], 0, 1000);
    }
    temp = temp << 1;
    if (sensorValue[j] > compare_value[j]) {
      temp |= 0x01;  // nếu cảm biến j có giá trị lớn hơn giá trị tham chiếu thì gán bit j của temp = 1
      // temp = temp |
    } else {
      temp &= 0xfe;
    }
    sensor = temp;
    Sum_sensor_value += sensorPDValue[j];
  }

  if (Sum_sensor_value > Out_line_sum) {
    //if (max_speed < inline_speed) max_speed++;

    for (int j = 0; j < 8; j++) {
      avg += (long)(sensorPDValue[j]) * ((j)*1000);
      sum += sensorPDValue[j];
    }
    error = (int)((avg / sum) - 3500);

    iP = kp * error;
    iD = kd * (lastPos - error);
    iRet = (iP - iD);
    if ((iRet < -4000)) {
      iRet = 0;
    }
    PD_Output = iRet / 30;  // 12 //25 //15 //20
    //PD_Output = iRet / 45;  // 12 //25 //15 //20

    lastPos = error;
  } else {
    max_speed = outline_speed;
  }
}
void trackLine(int speed) {
  robot_pid(PD_Output, speed);
}
void robot_pid(int diff, int speed) {
  int speedLeft;
  int speedRight;

  speedLeft = speed + diff;
  speedRight = speed - diff;

  // if robot inline => refresh speed
  motor_run(speedLeft, speedRight);
}

    if (sensorValue[i] > white_value[i])
      white_value[i] = sensorValue[i];
    compare_value[i] = (black_value[i] + white_value[i]) / 2;
  }
